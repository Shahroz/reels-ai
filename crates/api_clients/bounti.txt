window.addEventListener('load', function() {
  // Configuration Settings
  const config = {
    // Classes for targeting elements
    triggerClass: ".quote__________section_layout484", // The section that triggers the animation
    textClass: ".layout484_text",  // The text that will be split and animated

    // Scroll Trigger Points
    startTrigger: "top 80%",  // Animation starts when top of section reaches 80% up from bottom of viewport
    endTrigger: "bottom 20%", // Animation ends when bottom of section reaches 20% up from bottom of viewport
    showMarkers: false,        // Visual indicators for trigger points (helpful during development)

    // Animation Parameters
    initialOpacity: 0.3,      // Starting opacity of words (0 = invisible, 1 = fully visible)
    finalOpacity: 1,          // Ending opacity of words
    animationDuration: 0.8,   // How long the fade takes (in seconds)
    staggerTime: 0.1,         // Delay between each word's animation (in seconds)
    easeType: "power1.out"    // Animation easing function (controls acceleration/deceleration)
  };

  // Register ScrollTrigger plugin
  if (typeof ScrollTrigger !== "undefined") {
    gsap.registerPlugin(ScrollTrigger);
  }

  // Split text into individual words
  const layoutText = new SplitType(config.textClass, { types: "words" });

  // Set initial state
  gsap.set(layoutText.words, { opacity: config.initialOpacity });

  // Track animation state
  let isAnimating = false;
  let currentTimeline = null;

  // Animation function
  function animate(toFinal) {
    // If there's a current animation running, let it finish
    if (isAnimating) return;

    isAnimating = true;
    
    // Create a new timeline for this animation sequence
    currentTimeline = gsap.timeline({
      onComplete: () => {
        isAnimating = false;
        currentTimeline = null;
      }
    });

    if (toFinal) {
      // Staggered animation for appearing
      currentTimeline.to(layoutText.words, {
        opacity: config.finalOpacity,
        duration: config.animationDuration,
        stagger: config.staggerTime,
        ease: config.easeType
      });
    } else {
      // Instant reset for all words
      currentTimeline.to(layoutText.words, {
        opacity: config.initialOpacity,
        duration: config.animationDuration / 2, // Faster reset
        stagger: 0, // No stagger - all words together
        ease: "power1.in"
      });
    }
  }

  // Create ScrollTrigger instance
  const st = ScrollTrigger.create({
    trigger: config.triggerClass,
    start: config.startTrigger,
    end: config.endTrigger,
    markers: config.showMarkers,
    onEnter: () => {
      if (!isAnimating) animate(true);
    },
    onLeave: () => {
      if (!isAnimating) animate(false);
    },
    onEnterBack: () => {
      if (!isAnimating) animate(true);
    },
    onLeaveBack: () => {
      if (!isAnimating) animate(false);
    }
  });
});

function balanceHeadlines() {
  console.log('Balance headlines running');
  const headlines = document.querySelectorAll('.custom-2lines');
  console.log('Found headlines:', headlines.length);
  
  headlines.forEach((headline, index) => {
    console.log(`Processing headline ${index}:`, headline.innerText);
    const text = headline.innerText;
    const words = text.split(' ');
    
    // Get the actual height vs what it would be as single line
    const actualHeight = headline.offsetHeight;
    const singleLineHeight = parseInt(window.getComputedStyle(headline).lineHeight);
    console.log(`Heights - actual: ${actualHeight}, single line: ${singleLineHeight}`);
    
    // Only process if it's currently single line
    if (actualHeight <= singleLineHeight * 1.2) {  // allowing some margin for error
      // Find the middle point in terms of character count
      const totalChars = text.length;
      const midPoint = Math.floor(totalChars / 2);
      
      // Find the word break closest to the middle
      let bestBreakIndex = 0;
      let bestBreakDiff = totalChars;
      
      let charCount = 0;
      for (let i = 0; i < words.length - 1; i++) {
        charCount += words[i].length + 1;
        const diff = Math.abs(charCount - midPoint);
        if (diff < bestBreakDiff) {
          bestBreakDiff = diff;
          bestBreakIndex = i;
        }
      }
      
      // Insert break at the best point
      const firstLine = words.slice(0, bestBreakIndex + 1).join(' ');
      const secondLine = words.slice(bestBreakIndex + 1).join(' ');
      console.log(`Breaking "${text}" into:\n"${firstLine}" and\n"${secondLine}"`);
      headline.innerHTML = `${firstLine}<br>${secondLine}`;
    } else {
      console.log('Already properly multiline, skipping');
    }
  });
}

// Run after a slight delay to ensure rendering
document.addEventListener('DOMContentLoaded', () => {
  setTimeout(balanceHeadlines, 100);
});

// Also run on resize after a debounce
let resizeTimeout;
window.addEventListener('resize', () => {
  clearTimeout(resizeTimeout);
  resizeTimeout = setTimeout(balanceHeadlines, 100);
});
